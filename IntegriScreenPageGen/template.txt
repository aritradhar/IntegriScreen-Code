<script type="text/javascript">
  function calibrate()
    {
      // get current visible window size in different browsers
      var w = window,
        d = document,
        e = d.documentElement,
        g = d.getElementsByTagName('body')[0],
        x = w.innerWidth || e.clientWidth || g.clientWidth,
        y = w.innerHeight|| e.clientHeight|| g.clientHeight;

      //scale down the green frame
      url = new URL(location.href);
      scale = url.searchParams.get("scale");
      if (scale == null) {
        scale = 0.7;  // default value
        location = location+"?scale="+scale;
      }
      // add the scale in url to make it easier to change

      y=scale*y;

      // keep width:hight ratio to 16:9
      x = (!!height!!*y)/!!width!!;

      document.getElementById("frameBox").style.width = x+'px';
      document.getElementById("frameBox").style.height = y+'px';
    }
</script>
